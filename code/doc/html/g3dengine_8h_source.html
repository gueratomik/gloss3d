382  ( vec.y * vec.y ) + \
383  ( vec.z * vec.z ) ) )
384 
385 #define _FASTNORMALIZE(vec) \
386 { \
387  float number = ( (vec)->x * (vec)->x ) + \
388  ( (vec)->y * (vec)->y ) + \
389  ( (vec)->z * (vec)->z ); \
390  long i; \
391  float x2, y; \
392  const float threehalfs = 1.5f; \
393  \
394  x2 = number * 0.5f; \
395  y = number; \
396  i = * ( long * ) &y; \
397  i = 0x5f3759df - ( i >> 1 ); \
398  y = * ( float * ) &i; \
399  y = y * ( threehalfs - ( x2 * y * y ) ); \
400  \
401  (vec)->x = (vec)->x * y; \
402  (vec)->y = (vec)->y * y; \
403  (vec)->z = (vec)->z * y; \
404 }
